import flet as ft
import random
import json


def main(page: ft.Page):
    page.fonts = {"Jose": "Fonts/font4.ttf"}
    page.theme = ft.Theme(font_family="Jose")
    page.title = "EndoBotM Control Panel"
    page.window.resizable = False
    page.window.maximizable = False
    page.window.width = 1200
    page.window.height = 800
    # Fake client data
    clients = '{"client": 1, "hostname": "ENDOSPC", "ip": "192.168.1.1"}'

    client = json.loads(clients)
    # STATE
    showing_details = ft.Ref[bool]()
    showing_details.value = False
    selected_client = ft.Ref[str]()
    selected_client.value = None

    # Views
    def show_client_grid():
        client_grid.visible = True
        detail_view.visible = False
        page.update()

    def show_client_details(client_name):
        selected_client.value = client_name
        detail_title.value = f"{client_name} ‚Äî Control Center"
        client_grid.visible = False
        detail_view.visible = True
        page.update()

    # CLIENT GRID
    client_grid = ft.Container(
        content=ft.GridView(
            expand=1,
            runs_count=5,
            spacing=10,
            run_spacing=10,
            controls=[
                ft.ElevatedButton(
                    text=f"{int(client['client']) + i}",
                    on_click=lambda e, c=f"{int(client['client']) + i}": show_client_details(
                        c
                    ),
                    style=ft.ButtonStyle(
                        padding=10,
                        shape=ft.RoundedRectangleBorder(radius=8),
                    ),
                )
                for i in range(500)
            ],
        ),
        bgcolor=ft.Colors.INDIGO_900,
        height=650,
    )

    # DETAIL VIEW
    detail_title = ft.Text("", size=22, weight=ft.FontWeight.BOLD)

    def make_detail_tabs():
        return ft.Tabs(
            selected_index=0,
            animation_duration=200,
            tabs=[
                ft.Tab(
                    text="Remote Terminal",
                    icon=ft.Icons.TERMINAL_ROUNDED,
                    content=ft.Text("Terminal interface goes here."),
                ),
                ft.Tab(
                    text="Screen View",
                    icon=ft.Icons.REMOVE_RED_EYE,
                    content=ft.Text("Live screen stream view placeholder."),
                ),
                ft.Tab(
                    text="File Transfer",
                    icon=ft.Icons.DRIVE_FOLDER_UPLOAD_ROUNDED,
                    content=ft.Text("File management interface placeholder."),
                ),
                ft.Tab(
                    text="Detailed Info",
                    icon=ft.Icons.SCREEN_SEARCH_DESKTOP_ROUNDED,
                    content=ft.Text("Detailed client system info."),
                ),
                ft.Tab(
                    text="Misc. Features",
                    icon=ft.Icons.MISCELLANEOUS_SERVICES,
                    content=ft.Text("Random utilities placeholder."),
                ),
            ],
            expand=1,
        )

    back_button = ft.ElevatedButton(
        "Back to Clients",
        icon=ft.Icons.ARROW_BACK_ROUNDED,
        on_click=lambda e: show_client_grid(),
        style=ft.ButtonStyle(shape=ft.RoundedRectangleBorder(radius=8)),
    )

    detail_view = ft.Container(
        visible=False,
        content=ft.Column(
            controls=[
                back_button,
                detail_title,
                make_detail_tabs(),
            ],
            alignment=ft.MainAxisAlignment.START,
            horizontal_alignment=ft.CrossAxisAlignment.START,
        ),
    )

    # CLIENTS TAB
    clients_tab = ft.Tab(
        text="Clients",
        icon=ft.Icons.GROUPS_2_ROUNDED,
        content=ft.Container(
            content=ft.Column(
                controls=[
                    client_grid,
                    detail_view,
                ],
                expand=True,
            ),
            padding=10,
        ),
    )

    # HOME TAB
    home_tab = ft.Tab(
        text="Home",
        icon=ft.Icons.HOME_ROUNDED,
        content=ft.Container(
            padding=10,
            content=ft.Column(
                controls=[
                    ft.Text(
                        "Server Side Control Panel",
                        size=30,
                        weight=ft.FontWeight.BOLD,
                    ),
                    ft.Text(
                        "Made with lots of coffee by EndoBotM ü•Äüíî‚úåÔ∏è",
                        size=11,
                    ),
                    ft.Text(
                        "Help ya boi out by starring the repo ig...",
                        size=10,
                    ),
                ],
                alignment=ft.MainAxisAlignment.START,
                horizontal_alignment=ft.CrossAxisAlignment.START,
            ),
        ),
    )
    tabs = ft.Tabs(
        selected_index=0,
        animation_duration=200,
        tabs=[home_tab, clients_tab],
        expand=1,
    )

    page.add(tabs)
    page.update()


ft.app(target=main)
